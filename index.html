215easter_sunday <- function(year) {
216  a <- year %% 19; b <- year %/% 100; c <- year %% 100
217  d <- b %/% 4;    e <- b %% 4;      f <- (b + 8) %/% 25
218  g <- (b - f + 1) %/% 3

264# 2c. RETURN CALCULATION + STALE PRICE DIAGNOSTIC
265#
266# Returns are computed as adj_price / lag(adj_price) - 1  per ISIN.
267#
268# adj_price = prccd / ajexdi.  For Compustat Global (international), ajexdi
269# is the cumulative adjustment factor that adjusts for splits, stock dividends
270# AND cash dividends, so adj_price is already a total-return-equivalent series.
271# Using adj_price * trfd_clean would double-count cash dividends (+3.35% p.a.
272# for Norwegian stocks) — see diagnose_cw_vs_osebx.R for full proof.
273#
274# Fallback: when adj_price is unavailable, daily_ret_csv (the pre-computed
275# return column from the source CSV) is used instead.
276#
277# Stale carry-forwards: when Compustat carries the prior price into a
278# non-trading day, adj_price_t == adj_price_{t-1}, giving 0% return.
279# These rows are set to NA (spine is kept for later joins).
280#
281# A diagnostic CSV is written to disk for coverage review.
282# =============================================================================
283 
284# IMPORTANT — why trfd_clean is NOT used here:
285#
286# In Compustat Global (international), ajexdi is the cumulative adjustment
287# factor that accounts for splits, stock dividends AND cash dividends.
288# Therefore adj_price = prccd / ajexdi is already a total-return-equivalent
289# price series: on ex-dividend dates prccd falls by the dividend amount
290# while ajexdi falls proportionally, so adj_price stays continuous and
291# implicitly reinvests the dividend.
292#
293# Diagnostic proof (diagnose_cw_vs_osebx.R):
294#   - trfd_clean contributes +3.35% p.a. geometrically
295#   - On Equinor's ex-div date (2015-05-20): adj_price +0.13%,
296#     trfd +1.08% → adj_price*trfd formula gives +1.21% (double-counts dividend)
297#   - Cap-weighted portfolio using adj_price*trfd: 19.22% ann
298#   - Cap-weighted portfolio using adj_price only:  13.38% ann  ← correct
299#   - OSEBXGI benchmark:                            11.94% ann
300#   The remaining ~1.4% gap is explained by total vs free-float market cap
301#   (Equinor is 67% state-owned; our mktcap_nok uses total shares, so
302#   Equinor is over-weighted relative to OSEBX).

1613# Method note:
1614# Ledoit and Wolf show why covariance regularisation is essential in

2471perf <- function(rets, rf_vec, label, strat) {
2472  keep   <- !is.na(rets) & !is.na(rf_vec)
2473  rets   <- rets[keep]; rf_vec <- rf_vec[keep]
2474  n      <- length(rets)
2475  if (n < 20) return(NULL)

2477  ar     <- prod(1 + rets)^(TRADING_DAYS_PER_YEAR / n) - 1

2485             max_dd = min(cv / cummax(cv) - 1),
2486             skewness = sk, kurtosis = ku, n_days = n)
2487}
